**选择排序**
> 时间复杂度：O(n²)
```
function findSmallItem(arr) {
  <!-- 返回最小元素的索引和值 -->
  let smallIndex = 0
  for (let i in arr) {
    if (arr[i] < arr[smallIndex]) {
      smallIndex = i
    }
  }
  return smallIndex
}
function selectSort(arr) {
  const list = [], length = arr.length
  for (let i = 0; i < length; i++) {
    <!-- 将最小的元素加入新数组 并在入参中删除该元素 -->
    const smallIndex = findSmallItem(arr)
    list.push(arr[smallIndex])
    arr.splice(smallIndex, 1)
  }
  return list
}
```
