### 查找问题
**查找有无**
元素a是否存在 `Set` 集合

**Set**
* add()
* delete()
* has()
```javascript
let s = new Set()
s.add(1).add(2).add(2)
// 注意2被加入了两次

s.size // 2

s.has(1) // true
s.has(2) // true
s.has(3) // false

s.delete(2)
s.has(2) // false
```

**二分查找**
> 数组必须有序
```
<!-- 非递归 -->
function binary_search(list, key) {
  let low = 0,
      high = list.length - 1
  while (low <= high) {
    let mid = parseInt((low + high) / 2)
    if (list[mid] === key) {
      return mid
    } else if (list[mid] > key) {
      high = mid - 1
    } else if (list[mid] < key) {
      low = mid + 1
    } else {
      return -1
    }
  }
}
```
```
<!-- 递归 -->
function binary_search(list, low, high, key) {
  if (low > high) {
    return -1
  }
  let mid = parseInt((low + high) / 2)
  if (list[mid] === key) {
    return mid
  } else if (list[mid] > key) {
    high = mid - 1
    return binary_search(list, low, high, key)
  } else if (list[mid] < key) {
    low = mid + 1
    return binary_search(list, low, high, key)
  }
}
```
